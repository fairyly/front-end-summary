# 对象中键值重复判断

```
var arr = [
  {'type':'checkbox','name':'选项1'},
  {'type':'checkbox','name':'选项1'},
  {'type':'checkbox','name':'选项3'}
]

//判断 type 的值 和 name 值 是否有重复

var arrValues = arr.map(item=>item.name) // 返回数组

function isRepeat(arr){
  var hash = {};
  for(var i in arr) {
    if(hash[arr[i]]){
       return true;
     }
     hash[arr[i]] = true;
  }
  return false;
}
```
